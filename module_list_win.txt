# ディレクトリの作成
- name: Create directory [C:\work]
  win_file:
    path: C:\work
    state: directory

# ディレクトリの削除
- name: delete directory [C:\work]
  win_file:
    path: C:\work
    state: absent

# リモートホスト上でのファイルコピー
- name: Copy a single file where the source is on the remote host
  win_copy:
    src: C:\Temp\foo.txt
    dest: C:\ansible\foo.txt
    remote_src: yes

- name: copy OJDBC driver
